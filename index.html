import pygame import random pygame.init() # ====== MÀN HÌNH ====== WIDTH, HEIGHT = 500, 600 win = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Game Đua Xe") # ====== MÀU ====== WHITE = (255, 255, 255) GRAY = (60, 60, 60) GREEN = (0, 255, 0) RED = (255, 0, 0) # ====== XE NGƯỜI CHƠI ====== player = pygame.Rect(WIDTH//2 - 25, HEIGHT - 100, 50, 80) player_speed = 6 # ====== XE ĐỐI THỦ ====== enemies = [] for i in range(3): enemies.append( pygame.Rect(random.randint(60, WIDTH-110), random.randint(-600, -100), 50, 80) ) enemy_speed = 5 # ====== ĐIỂM ====== score = 0 font = pygame.font.SysFont(None, 40) clock = pygame.time.Clock() run = True game_over = False # ====== HÀM VẼ ====== def draw_road(): win.fill(GRAY) for i in range(0, HEIGHT, 40): pygame.draw.rect(win, WHITE, (WIDTH//2 - 5, i, 10, 20)) def draw_text(text, x, y): img = font.render(text, True, WHITE) win.blit(img, (x, y)) # ====== GAME LOOP ====== while run: clock.tick(60) for event in pygame.event.get(): if event.type == pygame.QUIT: run = False if game_over and event.type == pygame.KEYDOWN: if event.key == pygame.K_r: # Reset game score = 0 player.x = WIDTH//2 - 25 for e in enemies: e.y = random.randint(-600, -100) game_over = False if not game_over: draw_road() # ====== ĐIỀU KHIỂN ====== keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] and player.x > 50: player.x -= player_speed if keys[pygame.K_RIGHT] and player.x < WIDTH - 100: player.x += player_speed # ====== XE ĐỐI THỦ ====== for enemy in enemies: enemy.y += enemy_speed if enemy.y > HEIGHT: enemy.y = random.randint(-300, -100) enemy.x = random.randint(60, WIDTH-110) score += 1 if player.colliderect(enemy): game_over = True # ====== VẼ ====== pygame.draw.rect(win, GREEN, player) for enemy in enemies: pygame.draw.rect(win, RED, enemy) draw_text(f"Score: {score}", 10, 10) else: draw_text("GAME OVER", WIDTH//2 - 120, HEIGHT//2 - 40) draw_text("Press R to Restart", WIDTH//2 - 160, HEIGHT//2 + 10) pygame.display.update() pygame.quit()
